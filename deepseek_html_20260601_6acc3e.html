3. Kasus 2 – Object Pool / Caching

Rifqi Prayoga 251410005 SI2A

Factory constructor dengan cache untuk koneksi berdasarkan host:port.

// ==========================================
// Nama : Rifqi Prayoga
// NIM : 251410005
// Kelas : SI2A
// ==========================================

class ConnectionPool {
  final String host;
  final int port;

  static final Map _cache = {};

  ConnectionPool._(this.host, this.port);

  factory ConnectionPool(String host, int port) {
    var key = '$host:$port';

    if (_cache.containsKey(key)) {
      print('Returning cached connection for $key');
      return _cache[key]!;
    }

    print('Creating new connection for $key');
    var connection = ConnectionPool._(host, port);
    _cache[key] = connection;
    return connection;
  }

  void query(String sql) {
    print('Executing: $sql on $host:$port');
  }
}

void main() {
  var conn1 = ConnectionPool('localhost', 5432);
  conn1.query('SELECT * FROM users');
  print('');

  var conn2 = ConnectionPool('localhost', 5432);
  conn2.query('INSERT INTO products VALUES ("Laptop")');
  print('');

  var conn3 = ConnectionPool('db.example.com', 3306);
  conn3.query('UPDATE orders SET status="Shipped"');
  print('');

  print('conn1 dan conn2 sama? ${identical(conn1, conn2)}');
  print('conn1 dan conn3 sama? ${identical(conn1, conn3)}');
}

class AppConfig {
  final String environment;
  final String apiUrl;
  final bool debugMode;

  AppConfig._development()
      : environment = 'development',
        apiUrl = 'http://localhost:3000',
        debugMode = true;

  AppConfig._production()
      : environment = 'production',
        apiUrl = 'https://api.example.com',
        debugMode = false;

  AppConfig._staging()
      : environment = 'staging',
        apiUrl = 'https://staging.example.com',
        debugMode = true;

  factory AppConfig.fromEnvironment(String env) {
    switch (env.toLowerCase()) {
      case 'dev':
        return AppConfig._development();
      case 'prod':
        return AppConfig._production();
      case 'stage':
        return AppConfig._staging();
      default:
        return AppConfig._development();
    }
  }

  void printConfig() {
    print('Environment : $environment');
    print('API URL : $apiUrl');
    print('Debug Mode : $debugMode');
    print('------------------------');
  }
}

void main() {
  var dev = AppConfig.fromEnvironment('dev');
  var prod = AppConfig.fromEnvironment('prod');
  var stage = AppConfig.fromEnvironment('stage');

  dev.printConfig();
  prod.printConfig();
  stage.printConfig();
}

6. Factory dengan Named Constructors

Factory constructor fromJson, guest, dan create.

// ==========================================

// NIM : 251410005
// Kelas : SI2A
// ==========================================

class User {
  final String id;
  final String name;
  final String email;

  User._({
    required this.id,
    required this.name,
    required this.email,
  });

  factory User.fromJson(Map json) {
    return User._(
      id: json['id'],
      name: json['name'],
      email: json['email'],
    );
  }

  factory User.guest() {
    return User._(
      id: '0',
      name: 'Guest',
      email: 'guest@example.com',
    );
  }

  factory User.create(String name, String email) {
    return User._(
      id: '001',
      name: name,
      email: email,
    );
  }

  @override
  String toString() {
    return 'User(id: $id, name: $name, email: $email)';
  }
}

void main() {
  var user = User.fromJson({
    'id': '123',
    'name': 'Alice',
    'email': 'alice@example.com'
  });
  var guest = User.guest();
  var newUser = User.create('Bob', 'bob@example.com');

  print(user);
  print(guest);
  print(newUser);
}

10. Factory untuk Immutable Objects dengan Cache

Rifqi Prayoga 251410005 SI2A

Factory dengan cache untuk objek warna immutable.

// ==========================================
// Nama : Rifqi Prayoga
// NIM : 251410005
// Kelas : SI2A
// ==========================================

class Color {
  final int red;
  final int green;
  final int blue;

  static final Map _cache = {};

  Color._(this.red, this.green, this.blue);

  factory Color.rgb(int red, int green, int blue) {
    if (red < 0 || red > 255 || green < 0 || green > 255 || blue < 0 || blue > 255) {
      throw ArgumentError('RGB values must be 0-255');
    }
    var key = '$red,$green,$blue';
    if (!_cache.containsKey(key)) {
      _cache[key] = Color._(red, green, blue);
    }
    return _cache[key]!;
  }

  factory Color.red() => Color.rgb(255, 0, 0);
  factory Color.green() => Color.rgb(0, 255, 0);
  factory Color.blue() => Color.rgb(0, 0, 255);
  factory Color.white() => Color.rgb(255, 255, 255);
  factory Color.black() => Color.rgb(0, 0, 0);

  @override
  String toString() => 'Color($red, $green, $blue)';
}

void main() {
  var red1 = Color.red();
  var red2 = Color.rgb(255, 0, 0);

  print(red1);
  print(red2);
  print(red1 == red2);
}

11. Latihan – ConnectionFactory, NotificationFactory, dan ShapeFactory

Rifqi Prayoga 251410005 SI2A

Gabungan tiga factory: connection pool, notifikasi, shape dengan cache.

// ==========================================
// Nama : Rifqi Prayoga
// NIM : 251410005
// Kelas : SI2A
// ==========================================

// ConnectionFactory
class DatabaseConnection {
  final int id;
  DatabaseConnection(this.id);
  void connect() => print('Menggunakan koneksi database #$id');
}

class ConnectionFactory {
  static final List _pool = [];
  static DatabaseConnection getConnection() {
    if (_pool.isNotEmpty) return _pool.removeLast();
    return DatabaseConnection(DateTime.now().millisecondsSinceEpoch);
  }
  static void releaseConnection(DatabaseConnection conn) => _pool.add(conn);
}

// NotificationFactory
abstract class NotificationService {
  void send();
  factory NotificationService(String platform) {
    switch (platform.toLowerCase()) {
      case 'android': return AndroidNotification();
      case 'ios': return IOSNotification();
      case 'web': return WebNotification();
      default: throw ArgumentError('Platform tidak dikenal');
    }
  }
}
class AndroidNotification implements NotificationService {
  @override void send() => print('Mengirim notifikasi Android');
}
class IOSNotification implements NotificationService {
  @override void send() => print('Mengirim notifikasi iOS');
}
class WebNotification implements NotificationService {
  @override void send() => print('Mengirim notifikasi Web');
}

// ShapeFactory dengan Cache
class Shape {
  final String type;
  Shape._(this.type);
  static final Map _cache = {};
  factory Shape(String type) {
    if (!_cache.containsKey(type)) _cache[type] = Shape._(type);
    return _cache[type]!;
  }
  void draw() => print('Drawing $type');
}

void main() {
  print('=== ConnectionFactory ===');
  var conn1 = ConnectionFactory.getConnection();
  conn1.connect();
  ConnectionFactory.releaseConnection(conn1);
  var conn2 = ConnectionFactory.getConnection();
  conn2.connect();

  print('\n=== NotificationFactory ===');
  NotificationService('android').send();
  NotificationService('ios').send();
  NotificationService('web').send();

  print('\n=== ShapeFactory ===');
  var circle1 = Shape('circle');
  var circle2 = Shape('circle');
  circle1.draw();
  print('Object sama: ${circle1 == circle2}');
}

12. Challenge – AnimalFactory

Rifqi Prayoga 251410005 SI2A

Factory dengan cache untuk objek hewan (cat, dog, bird).

// ==========================================
// Nama : Rifqi Prayoga
// NIM : 251410005
// Kelas : SI2A
// ==========================================

abstract class Animal {
  final String name;
  Animal(this.name);
  void sound();

  static final Map _cache = {};

  factory Animal.create(String type, String name) {
    if (name.isEmpty) throw ArgumentError('Nama hewan tidak boleh kosong');
    String key = '$type-$name';
    if (_cache.containsKey(key)) return _cache[key]!;

    Animal animal;
    switch (type.toLowerCase()) {
      case 'cat': animal = Cat(name); break;
      case 'dog': animal = Dog(name); break;
      case 'bird': animal = Bird(name); break;
      default: throw ArgumentError('Jenis hewan tidak dikenal');
    }
    _cache[key] = animal;
    return animal;
  }
}

class Cat extends Animal {
  Cat(super.name);
  @override void sound() => print('$name : Meow');
}
class Dog extends Animal {
  Dog(super.name);
  @override void sound() => print('$name : Woof');
}
class Bird extends Animal {
  Bird(super.name);
  @override void sound() => print('$name : Tweet');
}

void main() {
  var cat1 = Animal.create('cat', 'Milo');
  var dog1 = Animal.create('dog', 'Buddy');
  var bird1 = Animal.create('bird', 'Rio');
  var cat2 = Animal.create('cat', 'Milo');

  cat1.sound();
  dog1.sound();
  bird1.sound();
  print('Cache bekerja: ${cat1 == cat2}');
}

13. Kesimpulan

Rifqi Prayoga 251410005 SI2A

Dari seluruh pembahasan mengenai Factory Constructor pada bahasa Dart, dapat disimpulkan bahwa:

- Factory constructor digunakan untuk mengontrol proses pembuatan objek sehingga lebih fleksibel dibandingkan constructor biasa.

- Factory constructor dapat mengembalikan instance dari class yang sama maupun subclass sesuai kebutuhan program.

- Factory constructor memungkinkan penerapan pola Factory Pattern untuk menyederhanakan proses pembuatan objek.

- Factory constructor dapat digunakan untuk melakukan validasi data sebelum objek dibuat sehingga meningkatkan konsistensi program.

- Factory constructor dapat memanfaatkan cache untuk menghemat memori dengan menggunakan kembali objek yang sudah ada.

- Factory constructor dapat dikombinasikan dengan Generic, Named Constructor, dan berbagai konsep Object-Oriented Programming (OOP).

- Seluruh konsep factory constructor membantu menghasilkan kode yang lebih efisien, terstruktur, mudah dipelihara, dan mudah dikembangkan.
